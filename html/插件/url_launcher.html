url_launcher

打开第三方应用的插件。

url_launcher: ^6.1.5


import 'package:url_launcher/url_launcher.dart';

Scaffold(
    appBar: AppBar(title: const Text('打开第三方应用')),
    body: Center(
        child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
                ElevatedButton(onPressed: _launchUrl, child: const Text('用浏览器打开网址')),
                ElevatedButton(onPressed: _launchMap, child: const Text('打开第三方APP')),
            ],
        ),
    ),
),

// 用浏览器打开网址
Future<void> _launchUrl() async {
    final Uri url = Uri.parse('https://www.163.com/');
    if (!await launchUrl(url)) {
        throw 'Could not launch $url';
    }
}

// 打开第三方APP
Future<void> _launchMap() async {
    // Android
    const url = 'geo:53.32,4.917';    // APP提供者的schema
    if (!await launchUrl(Uri.parse(url))) {
        // IOS
        // 苹果给地图提供了个url，通过这个url可以访问苹果手机里的地图软件
        const url = 'http://maps.apple.com/?ll=52.32,4.17';
        if (!await launchUrl(Uri.parse(url))) {
            throw 'Could not launch $url';
        }
    }
}
