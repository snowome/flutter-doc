import 'package:flutter/material.dart';
import 'package:flutter_swiper_view/flutter_swiper_view.dart';

class HomePage extends StatefulWidget {
  const HomePage({Key? key}) : super(key: key);

  @override
  State<HomePage> createState() => _HomePageState();
}

class _HomePageState extends State<HomePage> {
  final List _imageUrls = [
    'http://cms-bucket.ws.126.net/2022/0901/d1d3addep00rhisdc0166c000ak0057c.png?imageView&thumbnail=453y225&quality=85&type=webp',
    'http://cms-bucket.ws.126.net/2022/0901/372d1bcdj00rhidko0027c000cl0069c.jpg?imageView&thumbnail=453y225&quality=85&type=webp',
    'http://cms-bucket.ws.126.net/2022/0901/d4408951j00rhitup001oc000s600e3c.jpg?imageView&thumbnail=453y225&quality=85&type=webp',
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Center(
        child: Column(
          children: [
            SizedBox(
              height: 160,
              child: Swiper(
                itemBuilder: (context, index) {
                  return Image.network(
                    _imageUrls[index],
                    fit: BoxFit.fill,
                  );
                },
                itemCount: _imageUrls.length,
                autoplay: true,
                pagination: const SwiperPagination(),
                control: const SwiperControl(
                  color: Color.fromRGBO(255, 255, 255, 0.5),
                  size: 30,
                ),
              ),
            ),
          ],
        ),
      ),
    );
  }
}
