PageView组件

轮播图、左右滑动切换视图


final PageController _controller = PageController(
    // 初始打开哪个页面
    initialPage: 0,
);
// _controller.jumpToPage(index);            // PageView页面跳转方法

PageView(
    // PageView的控制类
    controller: _controller,
    // 页面切换时调用
    onPageChanged: (index) {},
    children: [],
    // 滚动的方向
    scrollDirection: Axis.horizontal,
    // 手势滚动逻辑：如下：
    // 不滚动：const NeverScrollableScrollPhysics(),
    // 总是滚动（默认）：const AlwaysScrollableScrollPhysics(),
    // 滚动到边缘时是否有bounce：const BouncingScrollPhysics(),
    physics: const AlwaysScrollableScrollPhysics(),
    // 设置为false禁用页面捕捉，对自定义滚动行为很有用。
    pageSnapping: true,
    // 是否反向滚动
    reverse: false,
),
