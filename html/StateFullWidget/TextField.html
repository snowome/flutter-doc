final TextEditingController _controller = TextEditingController();

@override
void initState() {
    if (widget.defaultText != null) {
        setState(() {
            _controller.text = widget.defaultText!;
        });
    }
    super.initState();
}

    controller: _controller,
    onChanged: (String text) {},
