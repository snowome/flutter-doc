BottomNavigationBar组件

底部导航栏


int _curentIndex = 0;

BottomNavigationBar(
    currentIndex: _curentIndex,
    onTap: (index) {
        setState(() {
            _curentIndex = index;
        });
        print(_curentIndex);
    },
    type: BottomNavigationBarType.fixed, // 不管tab选中没有,BottomNavigationBarItem里的label都显示
    items: [
        BottomNavigationBarItem(
            label: '主页',
            icon: Icon(
                Icons.home,
                color: Colors.grey,
            ),
            activeIcon: Icon(
                Icons.home,
                color: Colors.grey,
            ),
        ),
        BottomNavigationBarItem(
            label: '列表',
            icon: Icon(
                Icons.list,
                color: Colors.grey,
            ),
            activeIcon: Icon(
                Icons.list,
                color: Colors.grey,
            ),
        ),
    ],
),
