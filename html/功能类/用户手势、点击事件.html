用户手势、点击事件


Stack(
    children: [
        Column(
            children: [
                // 用户点击事件
                GestureDetector(
                    onTap: () => _printMsg('点击'),
                    onDoubleTap: () => _printMsg('双击'),
                    onLongPress: () => _printMsg('长按'),
                    onTapCancel: () => _printMsg('取消-手指离开了被点击范围。'),
                    onTapUp: (e) => _printMsg('松开-手指没有离开被点击范围。'),
                    onTapDown: (e) => _printMsg('按下'),
                    child: Container(
                        padding: const EdgeInsets.all(60),
                        decoration: const BoxDecoration(color: Colors.blue),
                        child: const Text(
                            '点我',
                            style: TextStyle(
                                fontSize: 36,
                                color: Colors.white,
                            ),
                        ),
                    ),
                ),
                Text(printString),
            ],
        ),
        // 用户手势->圆根据用户手指拖动的过程移动
        Positioned(
            left: moveX,
            top: moveY,
            child: GestureDetector(
                // 手指拖动过程中
                onPanUpdate: (e) => _doMove(e),
                // 一个圆
                child: Container(
                    width: 72,
                    height: 72,
                    decoration: BoxDecoration(
                        color: Colors.amber,
                        borderRadius: BorderRadius.circular(36),
                    ),
                ),
            )
        ),
    ],
),
